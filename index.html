function translateText() { const text = document.getElementById("textInput").value; const from = document.getElementById("fromLang").value; const to = document.getElementById("toLang").value; fetch("https://libretranslate.de/translate", { method: "POST", body: JSON.stringify({ q: text, source: from, target: to, format: "text" }), headers: { "Content-Type": "application/json" } }) .then(res => res.json()) .then(data => { document.getElementById("output").innerText = data.translatedText; }) .catch(error => { document.getElementById("output").innerText = "Error: " + error.message; }); }
